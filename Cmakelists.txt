cmake_minimum_required(VERSION 3.15)
project(ps-cli)

set(CMAKE_CXX_STANDARD 17)

find_package(Boost 1.74 REQUIRED)
find_package(OpenSSL REQUIRED)

add_executable(ps-cli src/main.cpp)

target_include_directories(ps-cli PRIVATE ${Boost_INCLUDE_DIRS} ${OPENSSL_INCLUDE_DIR})
target_link_libraries(ps-cli OpenSSL::SSL OpenSSL::Crypto pthread)
